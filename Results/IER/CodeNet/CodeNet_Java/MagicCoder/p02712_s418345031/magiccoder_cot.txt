        long ans = 0;
        for (int i = 1; i <= N; i++)
            if (i % 3 != 0 && i % 5 != 0)
                ans += i;
        out.println(ans);

        out.flush();
    }

    static class Input {
        private BufferedReader br;
        private String[] buff;
        private int index = 0;

        Input(InputStream is) {
            br = new BufferedReader(new InputStreamReader(is));
        }

            try {
                return br.readLine();

            return null;
